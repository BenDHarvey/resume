% vim: set textwidth=120:

% Example CV based on the 1.5-column-cv template. Main features:
% * uses the Roboto font family and IcoMoon icon set;
% * doesn't use colours, different font weights are used instead for styling;
% * because the CV fits on one page, header and footer is empty, since there isn't much useful info to put there;
% * includes a photo.
#+LATEX_HEADER: \documentclass[a4paper,10pt]{article}


% package imports
% ---------------

#+LATEX_HEADER: \usepackage[british]{babel} % for correct language and hyphenation and stuff
#+LATEX_HEADER: \usepackage{calc}           % for easier length calculations (infix notation)
#+LATEX_HEADER: \usepackage{enumitem}       % for configuring list environments
#+LATEX_HEADER: \usepackage{fancyhdr}       % for setting header and footer
%#+LATEX_HEADER: \usepackage{fontspec}       % for fonts
#+LATEX_HEADER: \usepackage{geometry}       % for setting margins (\newgeometry)
#+LATEX_HEADER: \usepackage{graphicx}       % for pictures
#+LATEX_HEADER: \usepackage{microtype}      % for microtypography stuff
#+LATEX_HEADER: \usepackage{xcolor}         % for colours


% margin and column widths
% ------------------------

% margins
#+LATEX_HEADER: \newgeometry{left=15mm,right=15mm,top=15mm,bottom=15mm}

% width of the gap between left and right column
#+LATEX_HEADER: \newlength{\cvcolumngapwidth}
#+LATEX_HEADER: \setlength{\cvcolumngapwidth}{3.5mm}

% left column width
#+LATEX_HEADER: \newlength{\cvleftcolumnwidth}
#+LATEX_HEADER: \setlength{\cvleftcolumnwidth}{44.5mm}

% right column width
#+LATEX_HEADER: \newlength{\cvrightcolumnwidth}
#+LATEX_HEADER: \setlength{\cvrightcolumnwidth}{\textwidth-\cvleftcolumnwidth-\cvcolumngapwidth}

% set paragraph indentation to 0, because it screws up the whole layout otherwise
#+LATEX_HEADER: \setlength{\parindent}{0mm}


% style definitions
% -----------------
% style categories explanation:
% * \cvnameXXX is used for the name;
% * \cvsectionXXX is used for section names (left column, accompanied by a horizontal rule);
% * \cvtitleXXX is used for job/education titles (right column);
% * \cvdurationXXX is used for job/education durations (left column);
% * \cvheadingXXX is used for headings (left column);
% * \cvmainXXX (and \setmainfont) is used for main text;
% * \cvruleXXX is used for the horizontal rules denoting sections.

% font families
#+LATEX_HEADER: \defaultfontfeatures{Ligatures=TeX} % reportedly a good idea, see https://tex.stackexchange.com/a/37251

#+LATEX_HEADER: \newfontfamily{\cvnamefont}{Roboto Medium}
#+LATEX_HEADER: \newfontfamily{\cvsectionfont}{Roboto Medium}
#+LATEX_HEADER: \newfontfamily{\cvtitlefont}{Roboto Regular}
#+LATEX_HEADER: \newfontfamily{\cvdurationfont}{Roboto Light Italic}
#+LATEX_HEADER: \newfontfamily{\cvheadingfont}{Roboto Regular}
#+LATEX_HEADER: \setmainfont{Roboto Light}

% colours
#+LATEX_HEADER: \definecolor{cvnamecolor}{HTML}{000000}
#+LATEX_HEADER: \definecolor{cvsectioncolor}{HTML}{000000}
#+LATEX_HEADER: \definecolor{cvtitlecolor}{HTML}{000000}
#+LATEX_HEADER: \definecolor{cvdurationcolor}{HTML}{000000}
#+LATEX_HEADER: \definecolor{cvheadingcolor}{HTML}{000000}
#+LATEX_HEADER: \definecolor{cvmaincolor}{HTML}{000000}
#+LATEX_HEADER: \definecolor{cvrulecolor}{HTML}{000000}

#+LATEX_HEADER: \color{cvmaincolor}

% styles
#+LATEX_HEADER: \newcommand{\cvnamestyle}[1]{{\Large\cvnamefont\textcolor{cvnamecolor}{#1}}}
#+LATEX_HEADER: \newcommand{\cvsectionstyle}[1]{{\normalsize\cvsectionfont\textcolor{cvsectioncolor}{#1}}}
#+LATEX_HEADER: \newcommand{\cvtitlestyle}[1]{{\large\cvtitlefont\textcolor{cvtitlecolor}{#1}}}
#+LATEX_HEADER: \newcommand{\cvdurationstyle}[1]{{\small\cvdurationfont\textcolor{cvdurationcolor}{#1}}}
#+LATEX_HEADER: \newcommand{\cvheadingstyle}[1]{{\normalsize\cvheadingfont\textcolor{cvheadingcolor}{#1}}}


% inter-item spacing
% ------------------

% vertical space after personal info and standard CV items
#+LATEX_HEADER: \newlength{\cvafteritemskipamount}
#+LATEX_HEADER: \setlength{\cvafteritemskipamount}{5mm plus 1.25mm minus 1.25mm}

% vertical space after sections
#+LATEX_HEADER: \newlength{\cvaftersectionskipamount}
#+LATEX_HEADER: \setlength{\cvaftersectionskipamount}{2mm plus 0.5mm minus 0.5mm}

% extra vertical space to be used when a section starts with an item with a heading (e.g. in the skills section),
% so that the heading does not follow the section name too closely
#+LATEX_HEADER: \newlength{\cvbetweensectionandheadingextraskipamount}
#+LATEX_HEADER: \setlength{\cvbetweensectionandheadingextraskipamount}{1mm plus 0.25mm minus 0.25mm}


% intra-item spacing
% ------------------

% vertical space after name
#+LATEX_HEADER: \newlength{\cvafternameskipamount}
#+LATEX_HEADER: \setlength{\cvafternameskipamount}{3mm plus 0.75mm minus 0.75mm}

% vertical space after personal info lines
#+LATEX_HEADER: \newlength{\cvafterpersonalinfolineskipamount}
#+LATEX_HEADER: \setlength{\cvafterpersonalinfolineskipamount}{2mm plus 0.5mm minus 0.5mm}

% vertical space after titles
#+LATEX_HEADER: \newlength{\cvaftertitleskipamount}
#+LATEX_HEADER: \setlength{\cvaftertitleskipamount}{1mm plus 0.25mm minus 0.25mm}

% value to be used as parskip in right column of CV items and itemsep in lists (same for both, for consistency)
#+LATEX_HEADER: \newlength{\cvparskip}
#+LATEX_HEADER: \setlength{\cvparskip}{0.5mm plus 0.125mm minus 0.125mm}

% set global list configuration (use parskip as itemsep, and no separation otherwise)
#+LATEX_HEADER: \setlist{parsep=0mm,topsep=0mm,partopsep=0mm,itemsep=\cvparskip}


% CV commands
% -----------

% creates a "personal info" CV item with the given left and right column contents, with appropriate vertical space after
% @param #1 left column content (should be the CV photo)
% @param #2 right column content (should be the name and personal info)
#+LATEX_HEADER: \newcommand{\cvpersonalinfo}[2]{
#+LATEX_HEADER:     % left and right column
#+LATEX_HEADER:     \begin{minipage}[t]{\cvleftcolumnwidth}
#+LATEX_HEADER:         \vspace{0mm} % XXX hack to align to top, see https://tex.stackexchange.com/a/11632
#+LATEX_HEADER:         \raggedleft #1
#+LATEX_HEADER:     \end{minipage}% XXX necessary comment to avoid unwanted space
#+LATEX_HEADER:     \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
#+LATEX_HEADER:     \begin{minipage}[t]{\cvrightcolumnwidth}
#+LATEX_HEADER:         \vspace{0mm} % XXX hack to align to top, see https://tex.stackexchange.com/a/11632
#+LATEX_HEADER:         #2
#+LATEX_HEADER:     \end{minipage}
#+LATEX_HEADER:
#+LATEX_HEADER:     % space after
#+LATEX_HEADER:     \vspace{\cvafteritemskipamount}
#+LATEX_HEADER: }

% typesets a name, with appropriate vertical space after
% @param #1 name text
#+LATEX_HEADER: \newcommand{\cvname}[1]{
#+LATEX_HEADER:     % name
#+LATEX_HEADER:     \cvnamestyle{#1}
#+LATEX_HEADER:
#+LATEX_HEADER:     % space after
#+LATEX_HEADER:     \vspace{\cvafternameskipamount}
#+LATEX_HEADER: }

% typesets a line of personal info beginning with an icon, with appropriate vertical space after
% @param #1 parameters for the \includegraphics command used to include the icon
% @param #2 icon filename
% @param #3 line text
#+LATEX_HEADER: \newcommand{\cvpersonalinfolinewithicon}[3]{
#+LATEX_HEADER:     % icon, vertically aligned with text (see https://tex.stackexchange.com/a/129463)
#+LATEX_HEADER:     \raisebox{.5\fontcharht\font`E-.5\height}{\includegraphics[#1]{#2}}
#+LATEX_HEADER:     % text
#+LATEX_HEADER:     #3
#+LATEX_HEADER:
#+LATEX_HEADER:     % space after
#+LATEX_HEADER:     \vspace{\cvafterpersonalinfolineskipamount}
#+LATEX_HEADER: }

% creates a "section" CV item with the given left column content, a horizontal rule in the right column, and with
% appropriate vertical space after
% @param #1 left column content (should be the section name)
#+LATEX_HEADER: \newcommand{\cvsection}[1]{
#+LATEX_HEADER:     % left and right column
#+LATEX_HEADER:     \begin{minipage}[t]{\cvleftcolumnwidth}
#+LATEX_HEADER:         \raggedleft\cvsectionstyle{#1}
#+LATEX_HEADER:     \end{minipage}% XXX necessary comment to avoid unwanted space
#+LATEX_HEADER:     \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
#+LATEX_HEADER:     \begin{minipage}[t]{\cvrightcolumnwidth}
#+LATEX_HEADER:         \textcolor{cvrulecolor}{\rule{\cvrightcolumnwidth}{0.3mm}}
#+LATEX_HEADER:     \end{minipage}
#+LATEX_HEADER:
#+LATEX_HEADER:     % space after
#+LATEX_HEADER:     \vspace{\cvaftersectionskipamount}
#+LATEX_HEADER: }

% creates a standard, multi-purpose CV item with the given left and right column contents, parskip set to cvparskip
% in the right column, and with appropriate vertical space after
% @param #1 left column content
% @param #2 right column content
#+LATEX_HEADER: \newcommand{\cvitem}[2]{
#+LATEX_HEADER:     % left and right column
#+LATEX_HEADER:     \begin{minipage}[t]{\cvleftcolumnwidth}
#+LATEX_HEADER:         \raggedleft #1
#+LATEX_HEADER:     \end{minipage}% XXX necessary comment to avoid unwanted space
#+LATEX_HEADER:     \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
#+LATEX_HEADER:     \begin{minipage}[t]{\cvrightcolumnwidth}
#+LATEX_HEADER:         \setlength{\parskip}{\cvparskip} #2
#+LATEX_HEADER:     \end{minipage}
#+LATEX_HEADER:
#+LATEX_HEADER:     % space after
#+LATEX_HEADER:     \vspace{\cvafteritemskipamount}
#+LATEX_HEADER: }

% typesets a title, with appropriate vertical space after
% @param #1 title text
#+LATEX_HEADER: \newcommand{\cvtitle}[1]{
#+LATEX_HEADER:     % title
#+LATEX_HEADER:     \cvtitlestyle{#1}
#+LATEX_HEADER:
#+LATEX_HEADER:     % space after
#+LATEX_HEADER:     \vspace{\cvaftertitleskipamount}
#+LATEX_HEADER:     % XXX need to subtract cvparskip here, because it is automatically inserted after the title "paragraph"
#+LATEX_HEADER:     \vspace{-\cvparskip}
#+LATEX_HEADER: }


% header and footer
% -----------------

% set empty header and footer
#+LATEX_HEADER: \pagestyle{empty}
